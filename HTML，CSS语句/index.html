HTML，CSS语句

<style>    /* CSS样式容器 */
    body{    /* 网页的身体部分 */
        margin: 0;    /*外边距*/
    }
    .header{    /*.指是是class*/
        background-color: #333;    /* background-color框体背景色 */
    }
    .container{
        width: 1226px;    /* width宽 */
        margin: 0 auto;    /* margin 0(上下) auto（左右） */
    }
    .header .menu{
        float: left;    /* float 浮动；left 靠左；right靠右 */
    }
    .header .account{
        float: right;
    }
    .header a{
        font-size: 15px;    /* 字体大小 */
        line-height: 40px;    /* 字体行高 */
        color: #b0b0b0;    /* 颜色 */
        display: inline-block;    /* display显示属性；inline行内；block独行；inline-block */
        margin-right: 10px;
        text-decoration: none;    /* 字体装饰 如 a 超链接下的横线 none为去掉 */
    }
    .header a:hover{    /* hover鼠标悬停 */
        color: #ffffff;
    }
    .header .app{
        position: relative;    /* position：relative 父*/
    }
    .header .app:hover .app-download{
        display: block;    /* display：block 显示：显示 */
    }
    .header .app .app-download{
        display: none;    /* display：none 显示：隐藏 */
        position: absolute;    /* position：absolute 子*/
    }
    .sub-header{
        height: 100px;
        text-align: center;    /* 文本对齐方式：中心 */
    }

    .sub-header .logo a img{    /* class:sub-header下class:logo下a标签下img标签 */
        height: 66px;
    }

    .sub-header .menu-list a{
        display: inline-block;
        padding: 0 10px;    /* 内边距 */
        color: #333;
        font-size: 16px;
        text-decoration: none;
    }
    .home .category{
        height: 460px;
        width: 234px;
        background: rgba(105,101,101,.6);    /* 框体背景色 */
        display: inline-block;
    }
    .home .category ul{
        height: 420px;
        margin: 0;
        padding: 20px 0;
        list-style-type: none;    /* 列表装饰：去掉 */
        background: rgba(105,101,101,.6);

        font-size: 14px;
        color: #000;
    }

    .classify .classify-menu ul li a{
        display: inline-block;
        padding-top: 30px;
        color: #fff;
        opacity: 0.5;    /* 透明度 */

    }
    .classify .classify-menu ul li a:hover{
        opacity: 1;    /* 透明度 */
    }
