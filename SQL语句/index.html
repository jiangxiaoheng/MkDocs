查

show create table tb1;

desc tb1;

增

create table tb1(id int,name char) default charset=utf8;

create table tb1(
    id int primary key auto_increment,  -- 主键（不允许为空，不允许重复）,自增（不传值且自维护）
    name varchar(16) not null,  -- 不允许为空
    age int null,   -- 允许为空（默认）
    lv int default 3    -- 插入数据时，lv列的值默认为3
) default charset=utf8; # 字符集

通常情况下，以下方式创建表【标准】

create table tb1(
    id int primary key auto_increment,
    name varchar(16) not null,
    age int null,
    lv int default 3
) default charset=utf8;

常用数据类型

  • tinyint

sql # tinyint 有符号，取值范围 -128 ~ 127 （有正有负）【默认】 # tinyint unsigned 无符号，取值范围 0 ~ 255 （只有正） create table t2( id int primary key auto_increment, name char, age tinyint, # 有符号：取值范围 -128 ~ 127 lv tinyint unsigned # 无符号：取值范围 0 ~ 255 ) default charset=utf8;

  • int

sql # int 有符号，取值范围： -2147483648 ~ 2147483647 # int unsigned 有符号，取值范围：0 ~ 4294967295

  • bigint

sql # bigint 有符号，取值范围： # bigint unsigned 有符号，取值范围：

  • float（不常用）

  • double（不常用）

  • decimal（更精准，常用）

sql # 精准的小数值，m是数字的总个数（负号不算），d是小数点后个数。m最大值为65，d最大值为30。 # 例如： create table tb3( id int primary key auto_increment, salary decimal(10,2) ) default charset=utf8;

  • char （速度快）

sql # 定长字符串，m代表字符串长度，最多可容纳255个字符。 char(11) # 定长字符串，固定字符进行存储。没有11个字符，也会按照11个储存 create table tb4( id int primary key auto_increment, mobile char(11) ) default charset=utf8;

  • varchar （节省空间)

sql # 变字符串，m代表字符串长度。最多可容纳 65535字节，中文则需要 /3 为最大m。 varchar(64) # 变字符串，真实数据长度进行存储 create table tb5( id int primary key auto_increment, mobile varchar(64) ) default charset=utf8;

  • text

sql # 存放最大长度为 65,535 个字符的字符串。 #通常，长文本会用text类型。例如：文章，新闻等。 create table tb6( id int primary key auto_increment, title varchar(64) content text ) default charset=utf8;

  • mediumtext （不太用）

存放最大长度为 16,777,215 个字符的字符串。

  • longtext （不太用）

用于 BLOBs (Binary Large OBjects)。存放最多 4,294,967,295 字节的数据。

  • datetime

sql 日期和时间的组合。格式：YYYY-MM-DD HH:MM:SS 注释：支持的范围是从 '1000-01-01 00:00:00' 到 '9999-12-31 23:59:59'

  • date

sql 日期。格式：YYYY-MM-DD 注释：支持的范围是从 '1000-01-01' 到 '9999-12-31'

案例一

create table tb7(
    id int primary key auto_increment,
    name varchar(64) not null,
    email varchar(64) not null,
    password varchar(64) not null,
    age tinyint,
    salary decimal(10,2),
    ctime date
) default charset=utf8;

insert into tb7(name,email,password,age,salary,ctime) value('hh','122522509@qq.com','root123',33,9999.99,'2014-12-12');
insert into tb7(name,email,password,age,salary,ctime) value('hh','122522509@qq.com','root123',null,null,null);

删

drop table tb1;

改

# 更改列属性 modify: 
alter table 表名 modify 列名 新类型
alter table test modify name_new int(10);

# 更改列名 change: 
alter table 表名 change 旧列名 新列名 类型
alter table test change name name_new varchar(32);

# 只更改列属性 
change: alter table 表名 列名 列名 新类型 相同的列名要写两次.
alter table test change  name_new name_new int(32);

操作表记录内容

查

select * from 数据库.表名称;  # 完整地址查询，在其他数据中也可查询
select * from 表名称;
select 列名称,列名称 from 表名称;
select 列名称,列名称 from 表名称 where 条件;   # 条件查询

select * from tb7;
select id,name from db2.tb7;
select * from db2.tb7;
select * from db2.tb7 where id>3;

增

insert into 表名(列,列) value(数据,数据),(数据,数据),(数据,数据);

insert into tb1(name,age) values('hengheng',33),('john',50),('jj',60);  # 插入多行数据

insert into tb1 values(1,'heng1'),(2,'xiao2'),(3,'jiang3'); # 按默认循序
insert into tb1(name,id) values(1,'heng1'),(2,'xiao2'),(3,'jiang3');    # 设置循序

删

delete from 表名;
delete from 表名 where 条件;

delete from tb7;    #删除表
delete from tb7 where id = 3;   # 条件删除
delete from tb7 where id = 3 and name = 'heng'; # 条件删除
delete from tb7 where id = 3 or name = 'hh';    # 条件删除
delete from tb7 where id in (1,5);  # 条件删除

改

update 表名 set 列=值, 列=值;
update 表名 set 列=值, where 条件;

update db2.tb7 set name='GOD';  # 批量修改
update db2.tb7 set name='heng' where id=3;  # 条件修改
update tb7 set age=age+1;
